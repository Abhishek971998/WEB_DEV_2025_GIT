Truthy & Falsy Values in JavaScript

Falsy Values (Only 7)

Falsy Value Example
false if (false) {}
0 if (0) {}
-0 if (-0) {}
"" (empty string) if ("") {}
null if (null) {}
undefined if (undefined) {}
NaN if (NaN) {}

Truthy Values (Everything Else)

  • Any non-zero number (e.g., 1, -1, 3.14, Infinity, -Infinity)
  • Any non-empty string (e.g., "hello", " ", "0")
  • Any object (e.g., {}, { key: "value" })
  • Any array (e.g., [], [1, 2, 3])
  • Any function (e.g., function() {})

Examples

if ("hello") console.log("Truthy!");  // ✅ Runs
if (" ") console.log("Truthy!");      // ✅ Runs (even a space is truthy)
if ("0") console.log("Truthy!");      // ✅ Runs (string "0" is truthy)
if ([]) console.log("Truthy!");       // ✅ Runs (empty array is truthy)
if ({}) console.log("Truthy!");       // ✅ Runs (empty object is truthy)
if (1) console.log("Truthy!");        // ✅ Runs
if (-1) console.log("Truthy!");       // ✅ Runs
if (Infinity) console.log("Truthy!"); // ✅ Runs
if (function() {}) console.log("Truthy!");  // ✅ Runs (function is truthy)
if (null) console.log("Falsy!");      // ❌ Doesn't run
if (0) console.log("Falsy!");        // ❌ Doesn't run
if (-0) console.log("Falsy!");       // ❌ Doesn't run
if ("") console.log("Falsy!");       // ❌ Doesn't run (empty string is falsy)
if (NaN) console.log("Falsy!");      // ❌ Doesn't run
if (undefined) console.log("Falsy!"); // ❌ Doesn't run
